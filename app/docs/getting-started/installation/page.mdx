## Import Styles

Import the package stylesheet once in your root layout, app entry, or app-level stylesheet.

```tsx copy
import "tiptap-react-ui/style.css";
```

## Basic Setup

### 1. Create an editor instance

Use the `useTiptapEditor` hook in a client component:

```tsx copy
"use client";

import { useTiptapEditor } from "tiptap-react-ui";

export default function MyEditor() {
  const { editor, isEditorLoading } = useTiptapEditor();

  if (isEditorLoading) return null;

  return (
    // Your editor component here
  );
}
```

`useTiptapEditor` defaults `immediatelyRender` to `false`, which helps avoid hydration errors in SSR frameworks like Next.js.

### 2. Render RichTextEditor

```tsx copy
"use client";

import { RichTextEditor, useTiptapEditor } from "tiptap-react-ui";
import type { OnSaveArgs } from "tiptap-react-ui";
import "tiptap-react-ui/style.css";

export default function MyEditor() {
  const { editor, isEditorLoading } = useTiptapEditor();

  if (isEditorLoading) return null;

  const handleSave = ({ json, html }: OnSaveArgs) => {
    console.log("Saved content(JSON):", json);
    console.log("Saved content(HTML):", html);
  };

  return (
    <RichTextEditor
      editor={editor}
      onSave={handleSave}
      enablePreview
      enableWordCount
    />
  );
}
```

### 3. Or render NotionEditor

```tsx copy
"use client";

import { NotionEditor, useTiptapEditor } from "tiptap-react-ui";
import "tiptap-react-ui/style.css";

export default function MyNotionEditor() {
  const { editor, isEditorLoading } = useTiptapEditor();

  if (isEditorLoading) return null;

  return <NotionEditor editor={editor} />;
}
```
