## Basic Setup



### 1. Create an editor instance

Use the `useTiptapEditor` hook in your component:

```jsx
import { useTiptapEditor } from 'tiptap-react-ui';

export default function MyEditor() {
  const {editor} = useTiptapEditor();

  return (
    // Your editor component here
  );
}
```
**Important**: When using **Next.js** or **React Server Components (RSC)**, you must set `immediatelyRender: false` in your editor configuration. This ensures the editor is rendered on the client only and helps avoid hydration errors.

ex: `useTiptapEditor({immediatelyRender: false}){:jsx}`

### 2. Render the editor

```jsx copy
import { RichTextEditor, useTiptapEditor } from "tiptap-react-ui";
import type {OnSaveArgs} from "tiptap-react-ui";
import "tiptap-react-ui/style.css"; // import rich text editor style

export default function MyEditor() {
  const {editor} = useTiptapEditor();

  const handleSave = (content:OnSaveArgs) => {
    const {json,html} = content
    console.log("Saved content(JSON):", json);
    console.log("Saved content(HTML):", html);

    // ...your code 
  };

  return (
    <RichTextEditor
      editor={editor}
      onSave={handleSave}
      enablePreview={true}
      immediatelyRenderPreview={false} // You must set this to false if you're using nextjs or rsc and enablePreview={true}
      enableWordCount={true}
    />
  );
}
```
